                          headers={'Authorization': f'Api-Key {key}', 'Content-Type': 'application/json'},
                          json={'modelUri': f'gpt://{fid}/yandexgpt-lite',
                                'completionOptions': {'temperature': 0.6, 'maxTokens': 1000},
                                'messages': [{'role': 'system', 'text': prompt}, {'role': 'user', 'text': q}]},
                          timeout=60)
        if r.status_code == 200:
            res = r.json()
            return res['result']['alternatives'][0]['message']['text'], None
        return None, f"–û—à–∏–±–∫–∞ {r.status_code}"
    except Exception as e:
        return None, str(e)
